<!-- On loading up the window's content, this JavaScript code stores the value
     of the previous page's (i.e. referrer) URL in the variable
     "feedbackPageUrl" in the browser's local storage. -->
<script>
  function start() {
    var ref = document.getElementById('referrer-url');
    ref.value = document.referrer;
    localStorage.setItem("feedbackPageUrl",ref.value);
  }
  window.onload = start;
</script>

     and the next <= 10) together in order to allow them to successfully submit
